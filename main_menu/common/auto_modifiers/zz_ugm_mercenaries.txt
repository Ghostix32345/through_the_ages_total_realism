# UGM mercenary supply layer.
# Applied to every country automatically (player, AI, and tags formed or released
# mid-game), so the scarce-pool model is parity-safe and always on. Kept in its own
# block rather than folded into the base rebalance so the supply cut reads cleanly
# next to the slower pool refill it pairs with.

ugm_mercenary_supply = {
	requires_real = no

	# Mercenary pool scarcity. The base merc availability bonus that every country
	# carries is trimmed so the recruitable pool a region holds is smaller, and a
	# region drained by a major mercenary war stays thin for longer. The cut is kept
	# modest so one-province minors can still hire a company in an emergency; it pairs
	# with the far slower pool refill to make a wiped host a real loss rather than a
	# host that can simply be re-bought next season.
	global_mercenaries_modifier = -0.20
}
