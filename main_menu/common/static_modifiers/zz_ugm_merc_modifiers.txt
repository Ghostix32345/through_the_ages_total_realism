# Mercenary clamp and reliance modifiers.
#
# ugm_merc_discount_floor compensates the cumulative hire-mercenary discount stack
#   so it can never drive the hire price below roughly 55% of its base. The vanilla
#   discount sources (Italy, advances, the nobles estate, government reforms) are
#   additive with no floor, so a country combining them all walks the discount down
#   past -0.90 and buys a doomstack for a fraction of its true cost. The monthly pulse
#   measures the net discount and re-grants this modifier at the exact strength needed
#   to bring the net back up to the floor, so honest hiring is untouched but the
#   stacked-discount exploit is removed. The +1.0 value is scaled by the pulse's
#   size parameter, which carries the measured overage.
ugm_merc_discount_floor = {
	game_data = { category = country }
	hire_mercenary_premium_cost_modifier = 1.0
}

# Reliance-on-mercenaries instability tiers. A realm that leans on hired companies
#   instead of its own levies pays for it in slowly eroding stability, so an
#   all-mercenary war machine carries a governing cost rather than being a strictly
#   better army. Applied by the yearly pulse in graduated bands keyed to how many
#   mercenary companies the country currently fields; a realm with only a company or
#   two is unaffected.
ugm_merc_reliance_light = {
	game_data = { category = country }
	stability_decay = 0.002
}

ugm_merc_reliance_heavy = {
	game_data = { category = country }
	stability_decay = 0.005
}
