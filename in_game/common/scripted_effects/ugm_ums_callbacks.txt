# UGM - UMS callback dispatcher.
# Fires once per setting change. var:ums_callback holds the flag of the
# interacted setting; each branch re-syncs that setting's global-variable alias
# so the matching yearly pulse reads the new value on its next evaluation.

ugm_handle_ums_callback = {
	switch = {
		trigger = var:ums_callback

		flag:ugm__historical_rails = {
			ums_sync_bool_alias = {
				setting = ugm__historical_rails
				alias = ugm__historical_rails
			}
		}

		flag:ugm__merc_reliance = {
			ums_sync_bool_alias = {
				setting = ugm__merc_reliance
				alias = ugm__merc_reliance
			}
		}
	}
}
