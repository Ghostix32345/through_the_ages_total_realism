# Mercenary clamp and reliance wiring.
#
# Both hooks are engine-fired country-scoped pulses. They are wired through
# on_actions = { ... } so they list-merge with the vanilla pulse and the other
# UGM sub-on-actions rather than replacing the pulse effect.

# Monthly: enforce the hire-discount floor (the stacked-discount exploit fix).
monthly_country_pulse = {
	on_actions = {
		ugm_merc_discount_floor_pulse
	}
}

# Yearly: refresh the mercenary-reliance instability tiers.
yearly_country_pulse = {
	on_actions = {
		ugm_merc_reliance_pulse
	}
}

# === Discount floor =========================================================
# Cheap outer trigger: only countries whose net hire-discount has actually
# walked past the floor pay the cost of the measurement. The modifier is removed
# first so the reading reflects only the vanilla discount stack, then re-granted
# at the exact strength needed to lift the net back to the floor (-0.45).
ugm_merc_discount_floor_pulse = {
	trigger = {
		is_subject = no
	}
	effect = {
		remove_country_modifier = ugm_merc_discount_floor
		if = {
			limit = {
				modifier:hire_mercenary_premium_cost_modifier < -0.45
			}
			add_country_modifier = {
				modifier = ugm_merc_discount_floor
				years = 1
				mode = replace
				size = ugm_merc_discount_floor_overage
			}
		}
	}
}

# === Reliance instability ===================================================
# Graduated by how many mercenary companies the realm currently fields. A realm
# with no hired companies is skipped by the outer trigger and never carries the
# modifier; the tiers are cleared and re-applied each year so a realm that
# stands its mercenaries down recovers immediately.
ugm_merc_reliance_pulse = {
	trigger = {
		has_global_variable = ugm__merc_reliance
		any_hired_mercenary = { count >= 1 }
	}
	effect = {
		remove_country_modifier = ugm_merc_reliance_light
		remove_country_modifier = ugm_merc_reliance_heavy
		if = {
			limit = { any_hired_mercenary = { count >= 6 } }
			add_country_modifier = {
				modifier = ugm_merc_reliance_heavy
				years = 2
				mode = replace
			}
		}
		else_if = {
			limit = { any_hired_mercenary = { count >= 3 } }
			add_country_modifier = {
				modifier = ugm_merc_reliance_light
				years = 2
				mode = replace
			}
		}
	}
}
