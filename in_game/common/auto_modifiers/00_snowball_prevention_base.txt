## Base Settings for Snowball Prevention System

# This modifier can be used to disable snowball prevention for specific nations
# If a nation has this modifier, they are exempt from snowball penalties
# Usage: add_country_modifier = { modifier = snowball_prevention_disabled years = 10 }
snowball_prevention_disabled = {
	# This is a flag - it has no effects
	# It just marks nations that should be exempt from snowball penalties
	potential_trigger = { always = no }  # Never naturally applies
}
