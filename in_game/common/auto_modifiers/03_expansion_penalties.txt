## Rapid Expansion Penalties
## Aggressively expanding nations face multiple penalties
## This discourages constant warmongeering and rewards consolidation

# Aggressive Expansion Impact (diplomatic isolation)
# NOTE: This re-enables the commented-out aggressive_expansion_mod from vanilla
aggressive_expansion_diplomatic_penalty = {
	potential_trigger = { 
		aggressive_expansion > 0
	}
	
	scales_with = {
		value = aggressive_expansion
		divide = 50  # Per 50 AE
		min = 0
		max = 1.0
	}
	
	# Diplomacy becomes harder
	cultural_influence = -0.02			# -2% per bracket (harder to culture)
	counter_espionage = -0.05			# -5% per bracket (more spies against you)
	spy_network_construction = -0.015	# -1.5% per bracket (harder to build networks)
	diplomatic_reputation = -0.05		# -0.05 per bracket (AI less likely to ally)
	accepted_allied_nation_interaction_duration = 0.1  # Alliances last shorter
}

# War Exhaustion Multiplier - lengthy wars are very costly
# Expanding powers face production penalties during conquest
war_exhaustion_expansion_penalty = {
	potential_trigger = {
		current_war_exhaustion > 10
		any_active_war = yes
	}
	
	scales_with = {
		value = current_war_exhaustion
		subtract = 10
		divide = 3
		min = 0
	}
	
	# War becomes increasingly ruinous  
	production_efficiency = -0.04		# -4% per WE bracket
	tax_revenue = -0.03					# -3% taxes per bracket
	trade_efficiency = -0.02			# -2% trade per bracket
	
	# Armies break down
	land_morale = -0.03					# Additional -3% morale per bracket
	naval_morale = -0.03				# Naval morale decays
	desertion_chance = 0.01				# +1% troops desert per bracket
}

# Unresolved Conquest Instability  
# Newly conquered provinces take time to integrate
recently_conquered_instability = {
	potential_trigger = {
		any_active_war = yes
	}
	
	scales_with = {
		value = number_of_recently_conquered_provinces  # Script value: provinces <5 yrs old
		divide = 10
		min = 0
		max = 0.5
	}
	
	# Conquered territories are liabilities, not assets, initially
	local_unrest = 0.5					# High unrest in new territories
	local_tax_base = -0.4				# -40% tax (territories devastated)
	
	# Need to garrison troops there
	garrison_size = 0.3					# +0.3 garrison per 10 provinces
}

# Overexpansion Event Trigger
# Nations expanding too fast invite coalitions and penalties
coalition_building_threshold = {
	potential_trigger = {
		coalition_grade_antagonism > 20  # Significant antagonism from neighbors
	}
	
	scales_with = {
		value = coalition_grade_antagonism
		divide = 10
		min = 0
		max = 1.0
	}
	
	# Coalition formation becomes more likely
	coalition_members_count = 0.1		# For every 10 antagonism = +1 member
	
	# Economic penalties from coalition threat
	diplomatic_reputation = -0.03		# Coalition partners avoid you
	trade_efficiency = -0.05			# Trade routes disrupted by hostile neighbors
	
	# Diplomatic crisis state
	stability_decay = 0.001				# +0.001 monthly decay
	global_unrest = 0.2					# Populations fear war
}
