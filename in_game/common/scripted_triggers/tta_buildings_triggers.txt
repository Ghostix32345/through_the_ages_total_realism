old_world_crops_trigger = {
	OR = {
		NOT = {
			culture = {
				is_culture_native_american = yes
			}
		}
		current_age_or_later = {
			age = age_6_revolutions
		}
	}
}

new_world_crops_trigger = {
	OR = {
		culture = {
			is_culture_native_american = yes
		}
		current_age_or_later = {
			age = age_6_revolutions
		}
	}
}
