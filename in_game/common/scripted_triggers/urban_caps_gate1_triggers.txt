# Gate 1 Urban Caps
# Rules:
# - Max 2 Towns per province
# - Max 1 City OR 1 Megapolis per province (mutually exclusive)
# - Combinations allowed: 2T, 1T+1C, 1T+1M
# - Hierarchy: Town -> City -> Megapolis. Nie überspringen!
# - Towns können parallel upgegradet werden.


# ============================================
# CAN UPGRADE TO TOWN - Gate 1 Rules
# ============================================
# Allowed if:
# - Province is empty
# - 1 Town exists (becomes 2T)
# - 1 City exists (becomes 1T+1C)
# - 1 Megapolis exists (becomes 1T+1M)

can_upgrade_to_town_under_province_cap = {
    exists = province
    province = {
        OR = {
            # Case 1: Province is empty (no towns, no cities, no megapolis)
            # This allows the first Town to be built
            AND = {
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:town
                    }
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:city
                    }
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:megalopolis
                    }
                }
            }

            # Case 2: 1 Town exists, no City, no Megapolis (becomes 2T)
            AND = {
                any_location_in_province = {
                    location_rank = location_rank:town
                    count = 1
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:city
                    }
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:megalopolis
                    }
                }
            }

            # Case 3: 1 City exists, no Town (becomes 1T+1C)
            AND = {
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:town
                    }
                }
                any_location_in_province = {
                    location_rank = location_rank:city
                    count = 1
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:megalopolis
                    }
                }
            }

            # Case 4: 1 Megapolis exists, no Town (becomes 1T+1M)
            AND = {
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:town
                    }
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:city
                    }
                }
                any_location_in_province = {
                    location_rank = location_rank:megalopolis
                    count = 1
                }
            }
        }
    }
}

# ============================================
# CAN UPGRADE TO CITY - Gate 1 Rules (HIERARCHY CHECK)
# ============================================
# Allowed ONLY if a Town exists. 
# NOT allowed if Province is empty.

can_upgrade_to_city_under_province_cap = {
    exists = province
    province = { 
        OR = {
            # Case 1: 1 Town exists, no City, no Megapolis
            # This Town becomes the City
            AND = {
                any_location_in_province = {
                    location_rank = location_rank:town
                    count = 1
                }
                any_location_in_province = {
                    location_rank = location_rank:town
                    count = 1
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:city
                    }
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:megalopolis
                    }
                }
            }

            # Case 2: 2 Towns exist, no City, no Megapolis
            # One of the Towns becomes the City (Result: 1T + 1C)
            AND = {
                any_location_in_province = {
                    location_rank = location_rank:town
                    count = 2
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:city
                    }
                }
                NOT = {
                    any_location_in_province = {
                        location_rank = location_rank:megalopolis
                    }
                }
            }
        }
    }
}

# ============================================
# CAN UPGRADE TO MEGALOPOLIS - Gate 1 Rules (HIERARCHY CHECK)
# ============================================
# Allowed ONLY if a City exists.
# NOT allowed if Province is empty or only Towns exist.

can_upgrade_to_megalopolis_under_province_cap = {
    exists = province
    province = {
        has_full_province_location_ownership = yes
        # Must have exactly one City
        any_location_in_province = {
            location_rank = location_rank:city
            count = 1
        }
        # Must NOT have any other City
        NOT = {
            any_location_in_province = {
                location_rank = location_rank:city
                count > 1
            }
        }
        # Must NOT have any Megapolis yet
        NOT = {
            any_location_in_province = {
                location_rank = location_rank:megalopolis
            }
        }
    }
}
