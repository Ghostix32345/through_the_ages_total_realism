governor_general = {
	audio_tier = 2
	is_foreign = no
	pop_type = nobles
	max_levels = 1
	important_for_UI = yes

	employment_size = large_noble_employment

	category = government_category

	rural_settlement = no
	town = yes
	city = yes
	megalopolis = no

	allow = {
		trigger_if = {
			limit = {
				scope:actor = {
					modifier:num_governor_generals <= 0
				}
			}
			custom_tooltip = {
				text = "Wir können noch keinen Generalgouverneur ernennen."
				always = no
			}
		}
		trigger_else = {
			custom_tooltip = {
				text = "Wir können keinen weiteren Generalgouverneur ernennen."
				scope:actor = {
					total_building_levels_including_construction:governor_general < modifier:num_governor_generals
				}
			}
		}
		owner = scope:actor #not in subjects
	}

	location_potential = {
		NOT = { is_connected_to_through_realm = owner.capital }
		is_overseas_for_owner = yes
		is_port = yes
	}

	remove_if = {
		OR = {
			location = { is_connected_to_through_realm = owner.capital }
			scope:actor = {
				total_building_levels_including_construction:governor_general > modifier:num_governor_generals
			}
		}
	}

	expensive = yes

	automation_build_allowed = no
	ai_forbid_shutdown = yes
	forbidden_for_estates = yes
	important_for_AI = yes

	build_time = government_build_time

	unique_production_methods = {
		naval_governor_maintenance = {
			tools = 1
			masonry = 1
			weaponry = 0.5
			paper = 3
			books = 1
			goods_gold = 1
			furniture = 2
			naval_supplies = 0.5
			fine_cloth = 0.5

			category = building_maintenance
		}
	}

	construction_demand = local_governor_construction

	modifier = {
		local_proximity_source = 100
		local_max_control = 1.0
	}
}
