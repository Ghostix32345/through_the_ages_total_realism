governor_general = {
	audio_tier = 2
	is_foreign = no
	pop_type = nobles
	max_levels = 1
	important_for_UI = yes

	employment_size = large_noble_employment

	category = government_category

	rural_settlement = no
	town = yes
	city = yes
	megalopolis = no

	allow = {
		custom_tooltip = {
			text = "Wir können in dieser Region keinen weiteren Generalgouverneur ernennen."
			scope:actor = {
				NOT = {
					any_owned_location = {
						filter = {
							region = root.region
						}

						NOT = { this = root }

						OR = {
							has_building = building_type:governor_general
							has_building_with_at_least_one_level = building_type:governor_general
						}
					}
				}
			}
		}

		owner = scope:actor #not in subjects
	}

	location_potential = {
		NOT = { is_connected_to_through_realm = scope:actor.capital }
		is_overseas_for_owner = yes
		is_port = yes

		NOT = { has_building = building_type:colonial_governor }
		NOT = { has_building_with_at_least_one_level = building_type:colonial_governor }
	}

	remove_if = {
		OR = {
			location = { is_connected_to_through_realm = owner.capital }
			location = { is_overseas_for_owner = no }
		}
	}

	expensive = yes

	automation_build_allowed = no
	ai_forbid_shutdown = yes
	forbidden_for_estates = yes
	important_for_AI = yes

	build_time = government_build_time

	unique_production_methods = {
		naval_governor_maintenance = {
			tools = 1
			masonry = 1
			weaponry = 0.5
			paper = 3
			books = 1
			goods_gold = 1
			furniture = 2
			naval_supplies = 0.5
			fine_cloth = 0.5

			category = building_maintenance
		}
	}

	construction_demand = local_governor_construction

	modifier = {
		local_proximity_source = 100
		local_max_control = 1.0
	}
}
